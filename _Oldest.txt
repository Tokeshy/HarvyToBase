    function viewLot(id, auction_id, revision_id) { 
        show_blockWin('<div class="ac"><img src="https://icetrade.by/i/preldr.gif" alt="Загрузка" width="32" height="32" /> <br />Идет загрузка информации, пожалуйста подождите.</div>', 'attn');

            'https://icetrade.by/lots/view',
            { 
                'id': id,
                'auction_id': auction_id,
                'revision_id': revision_id
            },
            function (data) { 
                $('#lotRow' + id).addClass('expanded');
                removeOverlay();
                $(data).insertAfter('#lotRow' + id);
            }
        );
    }

    function viewLots(auction_id, revision_id) { 
        show_blockWin('<div class="ac"><img src="https://icetrade.by/i/preldr.gif" alt="Загрузка" width="32" height="32" /> <br />Идет загрузка информации, пожалуйста подождите.</div>', 'attn');
        $.post(
            'https://icetrade.by/lots/viewLots',
            { 
                'auction_id': auction_id,
                'revision_id': revision_id
            },
            function (data) { 
                $('#lots_list').remove();
                removeOverlay();
                $(data).insertAfter('#lotsFrom');
                setBackgroundTable();
            }
        );
    }
